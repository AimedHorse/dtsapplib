63 static void clean_fwsocket(void *data) {
64  struct fwsocket *sock = data;
65 
66  if (sock->ssl) {
67  objunref(sock->ssl);
68  }
69 
70  /*im closing remove from parent list*/
71  if (sock->parent) {
72  if (sock->parent->children) {
73  remove_bucket_item(sock->parent->children, sock);
74  }
75  objunref(sock->parent);
76  }
77 
78  /*looks like the server is shut down*/
79  if (sock->children) {
80  objunref(sock->children);
81  }
82 
83  if (sock->sock >= 0) {
84  close(sock->sock);
85  }
86 }

109 static struct fwsocket *accept_socket(struct fwsocket *sock) {
110  struct fwsocket *si;
111  socklen_t salen = sizeof(si->addr);
112 
113  if (!(si = objalloc(sizeof(*si),clean_fwsocket))) {
114  return NULL;
115  }
116 
117  objlock(sock);
118  if ((si->sock = accept(sock->sock, &si->addr.sa, &salen)) < 0) {
119  objunlock(sock);
120  objunref(si);
121  return NULL;

125  si->proto = sock->proto;

129  }

135 static struct fwsocket *_opensocket(int family, int stype, int proto, const char *ipaddr, const char *port, void *ssl, int ctype, int backlog) {
136  struct addrinfo hint, *result, *rp;
137  struct fwsocket *sock = NULL;
138  socklen_t salen = sizeof(union sockstruct);
139 #ifndef __WIN32__
140  int on = 1;
141 #endif
142 
143  memset(&hint, 0, sizeof(hint));
144  hint.ai_family = family;
145  hint.ai_socktype = stype;
146  hint.ai_protocol = proto;
147 
148  if (getaddrinfo(ipaddr, port, &hint, &result) || !result) {
149  return (NULL);
150  }
151 
152  for(rp = result; rp; rp = result->ai_next) {
153  if (!(sock = make_socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol, ssl))) {
154  continue;
155  }
156  if ((!ctype && !connect(sock->sock, rp->ai_addr, rp->ai_addrlen)) ||
157  (ctype && !bind(sock->sock, rp->ai_addr, rp->ai_addrlen))) {
158  break;
159  }
160  objunref(sock);
161  sock = NULL;
162  }
163 
164  if (!sock || !rp) {
165  if (sock) {
166  objunref(sock);
167  }
168  freeaddrinfo(result);
169 
170  return (NULL);
171  }
172 
173  if (ctype) {
174  sock->flags |= SOCK_FLAG_BIND;
175  memcpy(&sock->addr.ss, rp->ai_addr, sizeof(sock->addr.ss));
176 #ifndef __WIN32__
177  setsockopt(sock->sock, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on));
178 #ifdef SO_REUSEPORT
179  setsockopt(sock->sock, SOL_SOCKET, SO_REUSEPORT, &on, sizeof(on));
180 #endif
181 #endif
182  switch(sock->type) {
183  case SOCK_STREAM:
184  case SOCK_SEQPACKET:
185  listen(sock->sock, backlog);
186  /* no break */
187  default:
188  break;
189  }
190  } else {
191  getsockname(sock->sock, &sock->addr.sa, &salen);
192  }
193 
194  freeaddrinfo(result);
195  return (sock);
196 }

236 static void *_socket_handler(void **data) {
237  struct socket_handler *sockh = *data;
238  struct fwsocket *sock = sockh->sock;
239  struct fwsocket *newsock;
240  struct timeval tv;
241  fd_set rd_set, act_set;
242  int selfd, sockfd, type, flags;
243  struct bucket_loop *bloop;
244 
245  objlock(sock);
246  FD_ZERO(&rd_set);
247  sockfd = sock->sock;
248  type = sock->type;
249  if ((sock->flags & SOCK_FLAG_BIND) && (sock->ssl || !(sock->type == SOCK_DGRAM))) {
250  flags = (SOCK_FLAG_BIND & sock->flags);
251  } else {
252  flags = 0;
253  }
254  FD_SET(sockfd, &rd_set);
255  objunlock(sock);
256 
257  while (framework_threadok(data) && !testflag(sock, SOCK_FLAG_CLOSE)) {
258  act_set = rd_set;
259  tv.tv_sec = 0;
260  tv.tv_usec = 20000;
261 
262  selfd = select(sockfd + 1, &act_set, NULL, NULL, &tv);
263 
264  /*returned due to interupt continue or timed out*/
265  if ((selfd < 0 && errno == EINTR) || (!selfd)) {
266  if ((type == SOCK_DGRAM) && (flags & SOCK_FLAG_BIND)) {
267  dtlshandltimeout(sock);
268  }
269  continue;
270  } else if (selfd < 0) {
271  break;
272  }
273 
274  if (FD_ISSET(sockfd, &act_set)) {
275  if (flags & SOCK_FLAG_BIND) {
276  switch (type) {
277  case SOCK_STREAM:
278  case SOCK_SEQPACKET:
279  newsock = accept_socket(sock);
280  break;
281  case SOCK_DGRAM:
282  newsock = dtls_listenssl(sock);
283  break;
284  default:
285  newsock = NULL;
286  break;
287  }
288  if (newsock) {
289  objref(sock);
290  newsock->parent = sock;
291  addtobucket(sock->children, newsock);
292  socketclient(newsock, sockh->data, sockh->client, NULL);
293  if (sockh->connect) {
294  sockh->connect(newsock, sockh->data);
295  }
296  objunref(newsock); /*pass ref to thread*/
297  }
298  } else {
299  sockh->client(sockh->sock, sockh->data);
300  }
301  }
302  }
303 
304  if (sock->ssl) {
305  ssl_shutdown(sock->ssl);
306  }
307 
308  /*close children*/
309  if (sock->children) {
310  bloop = init_bucket_loop(sock->children);
311  while(bloop && (newsock = next_bucket_loop(bloop))) {
312  remove_bucket_loop(bloop);
313  objlock(newsock);
314  if (newsock->parent) {
315  objunref(newsock->parent);
316  newsock->parent = NULL;
317  }
318  objunlock(newsock);
319  close_socket(newsock); /*remove ref*/
320  }
321  stop_bucket_loop(bloop);
322  }
323 
324  objunref(sock);
325 
326  return NULL;
327 }

329 static void _start_socket_handler(struct fwsocket *sock, socketrecv read,
330  socketrecv acceptfunc, threadcleanup cleanup, void *data) {
331  struct socket_handler *sockh;
332 
333  if (!sock || !read || !(sockh = objalloc(sizeof(*sockh), NULL))) {
334  return;
335  }
336 
337  sockh->sock = sock;
338  sockh->client = read;
339  sockh->cleanup = cleanup;
340  sockh->connect = acceptfunc;
341  sockh->data = data;
342 
343  /* grab ref for data and pass sockh*/
344  objref(data);
345  objref(sock);
346  framework_mkthread(_socket_handler, _socket_handler_clean, NULL, sockh);
347  objunref(sockh);
348 }

350 extern void socketserver(struct fwsocket *sock, socketrecv read,
351  socketrecv acceptfunc, threadcleanup cleanup, void *data) {
352 
353  objlock(sock);
354  if (sock->flags & SOCK_FLAG_BIND) {
355  if (sock->ssl || !(sock->type == SOCK_DGRAM)) {
356  sock->children = create_bucketlist(6, hash_socket);
357  }
358  if (sock->ssl && (sock->type == SOCK_DGRAM)) {
359  objunlock(sock);
360  dtsl_serveropts(sock);
361  } else {
362  objunlock(sock);
363  }
364  } else {
365  objunlock(sock);
366  }
367  _start_socket_handler(sock, read, acceptfunc, cleanup, data);
368 }
