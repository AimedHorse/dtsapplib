40 /*
41  * handle signals to cleanup gracefully on exit
42  */
43 static void framework_sig_handler(int sig, siginfo_t *si, void *unused) {
44  /* flag and clean all threads*/
45  switch (sig) {
46  case SIGUSR1:
47  case SIGUSR2:
48  case SIGHUP:
49  case SIGALRM:
50  if (!thread_signal(sig) && framework_core_info->sig_handler) {
51  framework_core_info->sig_handler(sig, si, unused);
52  }
53  break;
54  case SIGTERM:
55  case SIGINT:
56  default:
57  if (!thread_signal(sig)) {
58  if (framework_core_info->sig_handler) {
59  framework_core_info->sig_handler(sig, si, unused);
60  } else {
61  stopthreads(1);
62  exit(-1);
63  }
64  }
65  }
66 }
67 #endif

93 extern void daemonize() {
94 #ifndef __WIN32__

96 
97  /* fork and die daemonize*/
98  forkpid = fork();
99  if (forkpid > 0) {
100  /* im all grown up and can pass onto child*/
101  exit(0);
102  } else if (forkpid < 0) {
103  /* could not fork*/
104  exit(-1);
105  }
106

108  signal(SIGTSTP, SIG_IGN);

117 extern int lockpidfile(const char *runfile) {
118  int lck_fd = 0;
119 #ifndef __WIN32__
120  char pidstr[12];
121  pid_t mypid;
122 
123  mypid = getpid();
124  sprintf(pidstr,"%i\n", (int)mypid);
125  if (runfile && ((lck_fd = open(runfile, O_RDWR|O_CREAT, 0640)) > 0) && (!flock(lck_fd, LOCK_EX | LOCK_NB))) {
126  if (write(lck_fd, pidstr, strlen(pidstr)) < 0) {
127  close(lck_fd);
128  lck_fd = -1;
129  }
130  /* file was opened and not locked*/
131  } else if (runfile && lck_fd) {
132  close(lck_fd);
133  lck_fd = -1;
134  }
135 #endif
136  return (lck_fd);
137 }

203 extern void framework_mkcore(char *progname, char *name, char *email, char *web, int year, char *runfile, int flags, syssighandler sigfunc) {
204  struct framework_core *core_info;
205  if (framework_core_info) {
206  objunref(framework_core_info);
207  framework_core_info = NULL;
208  }

211  return;

242 extern int framework_init(int argc, char *argv[], frameworkfunc callback) {
243  struct framework_core *ci = framework_core_info;
244  int ret = 0;
245 
246  seedrand();
247  sslstartup();
248 
249  /*prinit out a GNU licence summary*/
250  if (!(ci->flags & FRAMEWORK_FLAG_NOGNU)) {
251  printgnu(ci->progname, ci->year, ci->developer, ci->email, ci->www);
252  }
253 
254  /* fork the process to daemonize it*/
255  if (ci->flags & FRAMEWORK_FLAG_DAEMON) {
256  daemonize();
257  }
258 
259  if ((ci->flock = lockpidfile(ci->runfile) < 0)) {
260  printf("Could not lock pid file Exiting\n");
261  objunref(ci);
262  return (-1);
263  }
264 
265 #ifndef __WIN32__
266  /* interupt handler close clean on term so physical is reset*/
267  configure_sigact(framework_core_info->sa);
268 #endif
269 
270  /*run the code from the application*/
271  if (callback) {
272  ret = callback(argc, argv);
273  /* wait for all threads to end*/
274  stopthreads(1);
275  }
276 
277  /* turn off the lights*/
278  objunref(ci);
279  return (ret);
280 }
