239  void (*func)(unsigned char *, const void *, unsigned long, const void *, unsigned long), short alglen) {
240  unsigned char okey[64], ikey[64];

382  * RFC 1701 Checksum based on code from the RFC

384 static uint16_t _checksum(const void *data, int len, const uint16_t check) {
385  uint64_t csum = 0;
386  const uint32_t *arr = (uint32_t *)data;
387 
388  /*handle 32bit chunks*/
389  while(len > 3) {
390  csum += *arr++;
391  len -= 4;
392  }
393 
394  /*handle left over 16 bit chunk*/
395  if (len > 1) {
396  csum += *(uint16_t *)arr;
397  arr = (uint32_t *)((uint16_t *)arr + 1);
398  len -= 2;
399  }
400 
401  /*handle odd byte*/
402  if (len) {
403  csum += *(uint8_t *)arr;
404  }
405 
406  /*add checksum when called as verify*/
407  if (check) {
408  csum += check;
409  }
410 
411  /*collapse to 16 bits adding all overflows leaving 16bit checksum*/
412  while(csum >> 16) {
413  csum = (csum & 0xffff) + (csum >> 16);
414  }
415 
416  return (~(uint16_t)csum);
417 }
