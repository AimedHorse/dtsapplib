41 /* Initialise a list head sructure
42  * malloc a chunk of data and init it
43  * sets next to NULL;
44  * sets prev to itself
45  * list data can be NULL
46  */
47 
48 #define LIST_INIT(head, entry) { \
49  if (!head && (!(head = malloc(sizeof(*head))))) { \
50  printf("Could not allocate memory for list head\n"); \
51  } else { \
52  head->data = entry; \
53  head->next = NULL; \
54  head->prev = (entry) ? head : NULL; \
55  } \
56 }

58 /*
59  * Add a element to a list head struct
60  * adds entry to tail
61  */
62 #define LIST_ADD(head, entry) { \
63  __typeof(head) _tmp_head = head; \
64  if (!_tmp_head) { \
65  LIST_INIT(_tmp_head, entry) \
66  head = _tmp_head; \
67  } else if (_tmp_head->prev) {\
68  __typeof(head) _ent_head = NULL; \
69  LIST_INIT(_ent_head, entry) \
70  if (_ent_head) { \
71  _ent_head->prev = _tmp_head->prev; \
72  _tmp_head->prev->next = _ent_head; \
73  _tmp_head->prev = _ent_head; \
74  } \
75  } else { \
76  _tmp_head->data = entry; \
77  _tmp_head->next = NULL; \
78  _tmp_head->prev = _tmp_head; \
79  } \
80 }

99 #define LIST_REMOVE_ENTRY(head, cur) { \
100  if (cur && (head == cur)) { \
101  cur->prev->next = NULL; \
102  if (cur->next) { \
103  cur->next->prev = cur->prev; \
104  } \
