152 static char *filterconf(const char *str, int minlen) {
153  char *tmp, *token;
154 
155  /*trim leading and trailing white space*/
156  tmp = trim(str);
157 
158  /*remove everything after the last # ignore if # is first*/
159  if ((token = strrchr(tmp, '#'))) {
160  if (token == tmp) {
161  return NULL;
162  }
163  token[0] = '\0';
164  }
165 
166  /*first char is #*/
167  if ((token = strchr(tmp, '#')) && (token == tmp)) {
168  return NULL;
169  }
170 
171  /*remove ; as first char*/
172  if ((token = strchr(tmp, ';')) && (token == tmp)) {
173  return NULL;
174  }
175 
176  /*too short*/
177  if (strlen(tmp) < minlen) {
178  return NULL;
179  }
180 
181  return (tmp);
182 }

184 extern int process_config(const char *configname, const char *configfile) {
185  struct config_file *file;
186  struct config_category *category = NULL;
187  FILE *config;
188  char line[256];
189  char item[128];
190  char value[128];
191  char *tmp = (char *)&line;
192  char *token;
193 
194  if (!configfiles) {
195  initconfigfiles();
196  }
197 
198  file = create_conf_file(configname, configfile);
199  addtobucket(configfiles, file);
200 
201  if (!(config = fopen(file->filepath, "r"))) {
202  return (-1);
203  }
204 
205  while(fgets(line, sizeof(line) - 1, config)) {
206  if (!(tmp = filterconf(line, 3))) {
207  continue;
208  }
209 
210  /*this is a new category*/
211  if ((token = strchr(tmp, '[')) && (token == tmp)) {
212  tmp++;
213  token = strrchr(tmp, ']');
214  token[0] = '\0';
215  tmp = trim(tmp);
216  if (!strlenzero(tmp)) {
217  if (category) {
218  objunref(category);
219  }
220  category = create_conf_category(tmp);
221  addtobucket(file->cat, category);
222  }
223  continue;
224  }
225 
226  if (sscanf(tmp, "%[^=] %*[=] %[^\n]", (char *)&item, (char *)&value) != 2) {
227  continue;
228  }
229 
230  if (!category) {
231  category = create_conf_category("default");
232  addtobucket(file->cat, category);
233  }
234 
235  add_conf_entry(category, trim(item), trim(value));
236  }
237  fclose(config);
238  if (category) {
239  objunref(category);
240  }
241  if (file) {
242  objunref(file);
243  }
244  return (0);
245 }
