101  struct ref_obj *ref;
102  int ret = 0;
103 
104  ptr = ptr - refobj_offset;
105  ref = (struct ref_obj*)ptr;
106 
107  if (!data || !ref || (ref->magic != REFOBJ_MAGIC)) {
108  return (ret);
109  }

111  /*double check just incase im gone*/

113  if ((ref->magic == REFOBJ_MAGIC) && (ref->cnt > 0)) {
114  ref->cnt++;
115  ret = ref->cnt;
116  }
117  pthread_mutex_unlock(ref->lock);
118  }
119 
120  return (ret);

253 /*
254  * a bucket list is a ref obj the "list" element is a
255  * array of "bucket" entries each has a hash
256  * the default is to hash the memory when there is no call back
257  */
258 extern void *create_bucketlist(int bitmask, blisthash hash_function) {
259  struct bucket_list *new;
260  short int buckets, cnt;
261 
262  buckets = (1 << bitmask);
263 
264  /* allocate session bucket list memory*/
265  if (!(new = objalloc(sizeof(*new) + (sizeof(void*) + sizeof(pthread_mutex_t) + sizeof(int)) * buckets, empty_buckets))) {
266  return NULL;
267  }
268 
269  /*initialise each bucket*/
270  new->bucketbits = bitmask;
271  new->list = (void *)((char*)new + sizeof(*new));
272  for (cnt = 0; cnt < buckets; cnt++) {
273  if ((new->list[cnt] = malloc(sizeof(*new->list[cnt])))) {
274  memset(new->list[cnt], 0, sizeof(*new->list[cnt]));
275  }
276  }
277 
278  /*next pointer is pointer to locks*/
279  new->locks = (void *)&new->list[buckets];
280  for (cnt = 0; cnt < buckets; cnt++) {
281  pthread_mutex_init(&new->locks[cnt], NULL);
282  }
283 
284  /*Next up version array*/
285  new->version = (void *)&new->locks[buckets];
286 
287  new->hash_func = hash_function;
288 
289  return (new);
290 }

292 static struct blist_obj *blist_gotohash(struct blist_obj *cur, unsigned int hash, int bucketbits) {
293  struct blist_obj *lhead = cur;
294 
295  if ((hash << bucketbits) < 0) {
296  do {
297  lhead = lhead->prev;
298  } while ((lhead->hash > hash) && lhead->prev->next);
299  } else {
300  while (lhead && lhead->next && (lhead->next->hash < hash)) {
301  lhead = lhead->next;
302  }
303  }
304 
305  return (lhead);
306 }

324 /*
325  * add a ref to the object for the bucket list
326  */
327 extern int addtobucket(struct bucket_list *blist, void *data) {
328  char *ptr = data;
329  struct ref_obj *ref;
330  struct blist_obj *lhead, *tmp;
331  unsigned int hash, bucket;
332 
333  if (!objref(blist)) {
334  return (0);
335  }
336 
337  if (!objref(data)) {
338  objunref(blist);
339  return (0);
340  }
341 
342  ptr = ptr - refobj_offset;
343  ref = (struct ref_obj*)ptr;
344 
345  hash = gethash(blist, data, 0);
346  bucket = ((hash >> (32 - blist->bucketbits)) & ((1 << blist->bucketbits) - 1));
347 
348  pthread_mutex_lock(&blist->locks[bucket]);
349  lhead = blist->list[bucket];
350  /*no head or non null head*/
351  if (!lhead || lhead->prev) {
352  if (!(tmp = malloc(sizeof(*tmp)))) {
353  pthread_mutex_unlock(&blist->locks[bucket]);
354  objunref(data);
355  objunref(blist);
356  return (0);
357  }
358  memset(tmp, 0, sizeof(*tmp));
359  tmp->hash = hash;
360  tmp->data = ref;
361 
362  /*there is no head*/
363  if (!lhead) {
364  blist->list[bucket] = tmp;
365  tmp->prev = tmp;
366  tmp->next = NULL;
367  /*become new head*/
368  } else if (hash < lhead->hash) {
369  tmp->next = lhead;
370  tmp->prev = lhead->prev;
371  lhead->prev = tmp;
372  blist->list[bucket] = tmp;
373  /*new tail*/
374  } else if (hash > lhead->prev->hash) {
375  tmp->prev = lhead->prev;
376  tmp->next = NULL;
377  lhead->prev->next = tmp;
378  lhead->prev = tmp;
379  /*insert entry*/
380  } else {
381  lhead = blist_gotohash(lhead, hash, blist->bucketbits);
382  tmp->next = lhead->next;
383  tmp->prev = lhead;
384 
385  if (lhead->next) {
386  lhead->next->prev = tmp;
387  } else {
388  blist->list[bucket]->prev = tmp;
389  }
390  lhead->next = tmp;
391  }
392  } else {
393  /*set NULL head*/
394  lhead->data = ref;
395  lhead->prev = lhead;
396  lhead->next = NULL;
397  lhead->hash = hash;
398  }
399 
400  blist->version[bucket]++;
401  pthread_mutex_unlock(&blist->locks[bucket]);
402 
403  objlock(blist);
404  blist->count++;
405  objunlock(blist);
406  objunref(blist);
407 
408  return (1);
409 }

444 /*
445  * return the next object (+ref) in the list
446  */
447 extern void *next_bucket_loop(struct bucket_loop *bloop) {
448  struct bucket_list *blist = bloop->blist;
449  struct ref_obj *entry = NULL;
450  void *data = NULL;
451 
452  pthread_mutex_lock(&blist->locks[bloop->bucket]);
453  if (bloop->head_hash && (blist->version[bloop->bucket] != bloop->version)) {
454  /* bucket has changed unexpectedly i need to ff/rew to hash*/
455  bloop->head = blist_gotohash(blist->list[bloop->bucket], bloop->head_hash + 1, blist->bucketbits);
456  /*if head has gone find next suitable ignore any added*/
457  while (bloop->head && (bloop->head->hash < bloop->head_hash)) {
458  bloop->head = bloop->head->next;
459  }
460  }
461 
462  while (!bloop->head || !bloop->head->prev) {
463  pthread_mutex_unlock(&blist->locks[bloop->bucket]);
464  bloop->bucket++;
465  if (bloop->bucket < (1 << blist->bucketbits)) {
466  pthread_mutex_lock(&blist->locks[bloop->bucket]);
467  bloop->head = blist->list[bloop->bucket];
468  } else {
469  return NULL;
470  }
471  }
472 
473  if (bloop->head) {
474  bloop->cur = bloop->head;
475  entry = (bloop->head->data) ? bloop->head->data : NULL;
476  data = (entry) ? entry->data : NULL;
477  objref(data);
478  bloop->head = bloop->head->next;
479  bloop->head_hash = (bloop->head) ? bloop->head->hash : 0;
480  bloop->cur_hash = (bloop->cur) ? bloop->cur->hash : 0;
481  }
482  pthread_mutex_unlock(&blist->locks[bloop->bucket]);
483 
484  return (data);
485 }

487 extern void remove_bucket_item(struct bucket_list *blist, void *data) {
488  struct blist_obj *entry;
489  int hash, bucket;
490 
491  hash = gethash(blist, data, 0);
492  bucket = ((hash >> (32 - blist->bucketbits)) & ((1 << blist->bucketbits) - 1));
493 
494  pthread_mutex_lock(&blist->locks[bucket]);
495  entry = blist_gotohash(blist->list[bucket], hash + 1, blist->bucketbits);
496  if (entry && entry->hash == hash) {
497  if (entry->next && (entry == blist->list[bucket])) {
498  entry->next->prev = entry->prev;
499  blist->list[bucket] = entry->next;
500  } else if (entry->next) {
501  entry->next->prev = entry->prev;
502  entry->prev->next = entry->next;
503  } else if (entry == blist->list[bucket]) {
504  blist->list[bucket] = NULL;
505  } else {
506  entry->prev->next = NULL;
507  blist->list[bucket]->prev = entry->prev;
508  }
509  objunref(entry->data->data);
510  free(entry);
511  }
512  pthread_mutex_unlock(&blist->locks[bucket]);
513 }

515 /*
516  * remove and unref the current data
517  */
518 extern void remove_bucket_loop(struct bucket_loop *bloop) {
519  struct bucket_list *blist = bloop->blist;
520  int bucket = bloop->bucket;
521 
522  pthread_mutex_lock(&blist->locks[bloop->bucket]);
523  /*if the bucket has altered need to verify i can remove*/
524  if (bloop->cur_hash && (!bloop->cur || (blist->version[bloop->bucket] != bloop->version))) {
525  bloop->cur = blist_gotohash(blist->list[bloop->bucket], bloop->cur_hash + 1, blist->bucketbits);
526  if (!bloop->cur || (bloop->cur->hash != bloop->cur_hash)) {
527  pthread_mutex_unlock(&blist->locks[bucket]);
528  return;
529  }
530  }
531 
532  if (!bloop->cur) {
533  pthread_mutex_unlock(&blist->locks[bucket]);
534  return;
535  }
536 
537  if (bloop->cur->next && (bloop->cur == blist->list[bucket])) {
538  bloop->cur->next->prev = bloop->cur->prev;
539  blist->list[bucket] = bloop->cur->next;
540  } else if (bloop->cur->next) {
541  bloop->cur->next->prev = bloop->cur->prev;
542  bloop->cur->prev->next = bloop->cur->next;
543  } else if (bloop->cur == blist->list[bucket]) {
544  blist->list[bucket] = NULL;
545  } else {
546  bloop->cur->prev->next = NULL;
547  blist->list[bucket]->prev = bloop->cur->prev;
548  }
549 
550  objunref(bloop->cur->data->data);
551  free(bloop->cur);
552  bloop->cur_hash = 0;
553  bloop->cur = NULL;
554  blist->version[bucket]++;
555  bloop->version++;
556  pthread_mutex_unlock(&blist->locks[bucket]);
557 
558  objlock(blist);
559  blist->count--;
560  objunlock(blist);
561 }
