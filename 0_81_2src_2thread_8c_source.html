108 /*
109  * loop through all threads till they stoped
110  * setting stop will flag threads to stop
111  */
112 static void *managethread(void **data) {
113  struct thread_pvt *mythread = threads->manager;
114  struct thread_pvt *thread;
115  struct bucket_loop *bloop;
116  pthread_t me;
117  int stop = 0;
118 
119  me = pthread_self();
120  while(bucket_list_cnt(threads->list)) {
121  bloop = init_bucket_loop(threads->list);
122  while (bloop && (thread = next_bucket_loop(bloop))) {
123  /*this is my call im done*/
124  if (pthread_equal(thread->thr, me)) {
125  /* im going to leave the list and try close down all others*/
126  if (mythread && !(testflag(mythread, TL_THREAD_RUN))) {
127  remove_bucket_loop(bloop);
128  stop = 1;
129  }
130  objunref(thread);
131  continue;
132  }
133 
134  objlock(thread);
135  if (stop && (thread->flags & TL_THREAD_RUN) && !(thread->flags & TL_THREAD_DONE)) {
136  thread->flags &= ~TL_THREAD_RUN;
137  objunlock(thread);
138  } else if ((thread->flags & TL_THREAD_DONE) || pthread_kill(thread->thr, 0)){
139  objunlock(thread);
140  remove_bucket_loop(bloop);
141  if (thread->cleanup) {
142  thread->cleanup(thread->data);
143  }
144  objunref(thread->data);
145  objunref(thread);
146  } else {
147  objunlock(thread);
148  }
149  objunref(thread);
150  }
151  stop_bucket_loop(bloop);
152 #ifdef __WIN32__
153  Sleep(100);
154 #else
155  sleep(1);
156 #endif
157  }
158  objunref(threads);
159 
160  return NULL;
161 }

191 static void *threadwrap(void *data) {
192  struct thread_pvt *thread = data;
193  void *ret = NULL;
194 
195  if (thread && thread->func) {
196  setflag(thread, TL_THREAD_RUN);
197  ret = thread->func(&thread->data);
198  setflag(thread, TL_THREAD_DONE);
199  }
200 
201  if (!threads || !threads->manager) {
202  if (thread->cleanup) {
203  thread->cleanup(thread->data);
204  }
205  objunref(thread->data);
206  objunref(thread);
207  }
208 
209  return (ret);
210 }

212 /*
213  * create a thread
214  */
215 extern struct thread_pvt *framework_mkthread(threadfunc func, threadcleanup cleanup, threadsighandler sig_handler, void *data) {
216  struct thread_pvt *thread;
217 
218  if (!(thread = objalloc(sizeof(*thread), NULL))) {
219  return NULL;
220  }
221 
222  thread->data = data;
223  thread->flags = 0;
224  thread->cleanup = cleanup;
225  thread->sighandler = sig_handler;
226  thread->func = func;
227  thread->magic = THREAD_MAGIC;
228 
229  /* grab a ref to data for thread to make sure it does not go away*/
230  objref(thread->data);
231  if (pthread_create(&thread->thr, NULL, threadwrap, thread)) {
232  objunref(thread);
233  objunref(thread->data);
234  return NULL;
235  }
236 
237  /* am i up and running move ref to list*/
238  if (!pthread_kill(thread->thr, 0)) {
239  if ((threads && threads->list) || startthreads()) {
240  objlock(threads);
241  addtobucket(threads->list, thread);
242  objunlock(threads);
243  }
244  return (thread);
245  } else {
246  objunref(thread->data);
247  objunref(thread);
248  }
249 
250  return NULL;
251 }

270 /*
271  * find the thread the signal was delivered to
272  * if the signal was handled returns 1
273  * if the thread could not be handled returns -1
274  * returns 0 if not for thread
275  * NB sending a signal to the current thread while threads is locked
276  * will cause a deadlock.
277  */
278 extern int thread_signal(int sig) {
279  struct thread_pvt *thread;
280  pthread_t me;
281  int ret = 0;
282 
283  me = pthread_self();
284  if ((thread = bucket_list_find_key(threads->list, &me))) {
285  if (thread->sighandler) {
286  thread->sighandler(sig, thread);
287  ret = 1;
288  } else {
289  ret = -1;
290  }
291  objunref(thread);
292  }
293  return (ret);
294 }
