149 extern void _hmac(unsigned char *buff, const void *data, unsigned long len, const void *key, unsigned long klen,
150  hmacfunc func, short alglen) {
151  unsigned char okey[64], ikey[64];
152  int bcnt;
153 
154  memset(ikey, 0, 64);
155  memset(okey, 0, 64);
156 
157  if (klen < 64) {
158  memcpy(ikey, key, klen);
159  memcpy(okey, key, klen);
160  } else {
161  md5sum(okey, key, klen);
162  memcpy(ikey, okey, klen);
163  }
164 
165  for (bcnt = 0; bcnt < 64; bcnt++) {
166  ikey[bcnt] ^= 0x36;
167  okey[bcnt] ^= 0x5c;
168  };
169 
170  func(buff, ikey, 64, data, len);
171  func(buff, okey, 64, buff, alglen);
172 }

240 /*
241  * RFC 1701 Checksum based on code from the RFC
242  */
243 extern uint16_t _checksum(const void *data, int len, const uint16_t check) {
244  uint64_t csum = 0;
245  const uint32_t *arr = (uint32_t*)data;
246 
247  /*handle 32bit chunks*/
248  while(len > 3) {
249  csum += *arr++;
250  len -= 4;
251  }
252 
253  /*handle left over 16 bit chunk*/
254  if (len > 1) {
255  csum += *(uint16_t*)arr;
256  arr = (uint32_t*)((uint16_t*)arr + 1);
257  len -= 2;
258  }
259 
260  /*handle odd byte*/
261  if (len) {
262  csum += *(uint8_t*)arr;
263  }
264 
265  /*add checksum when called as verify*/
266  if (check) {
267  csum += check;
268  }
269 
270  /*collapse to 16 bits adding all overflows leaving 16bit checksum*/
271  while(csum >> 16) {
272  csum = (csum & 0xffff) + (csum >> 16);
273  }
274 
275  return (~(uint16_t)csum);
276 }

308 extern char *b64enc_buf(const char *message, uint32_t len, int nonl) {
309  BIO *bmem, *b64;
310  BUF_MEM *ptr;

313

318  bmem = BIO_new(BIO_s_mem());
319  b64 = BIO_push(b64, bmem);

321  BIO_set_flags(b64, BIO_FLAGS_BASE64_NO_NL);
322  }

327  buffer = objalloc(ptr->length+1, NULL);

330 
331  BIO_free_all(b64);

334 }
